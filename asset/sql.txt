    V int(11) DEFAULT 0,
    O int(11) DEFAULT 0,
    F int(11) DEFAULT 0,
    GM int(11) DEFAULT 0,
    IM int(11) DEFAULT 0,
    Diff int(11) DEFAULT 0,
    P int(11) DEFAULT 0,

);

    score1 int(11),
    score2 int(11)
);

	id int(11) not null AUTO_INCREMENT PRIMARY KEY,
    nummer int(11),
    spelare varchar(128) not null,
    program varchar(128) not null,
    team varchar(128) not null
);
